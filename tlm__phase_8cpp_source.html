33struct tlm_phase_registry
34{
35 typedef unsigned int key_type;
36
37 static tlm_phase_registry& instance()
38 { static tlm_phase_registry inst; return inst; }
39
40 unsigned int register_phase(sc_type_index type, sc_string_view name)
41 {
42 type_map::const_iterator it = ids_.find( type );
43
44 if( name.empty() ) {
45 SC_REPORT_FATAL( sc_core::SC_ID_INTERNAL_ERROR_,
46 "unexpected empty tlm_phase name" );
47 return UNINITIALIZED_PHASE;
48 }
49
50 if( it == ids_.end() ) { // new phase - generate/store ID and name
51 type_map::value_type v( type, static_cast<key_type>(names_.size()) );
52 names_.push_back( name_table::value_type(name.data(), name.size()) );
53 ids_.insert( v );
54 return v.second;
55 }
56
57 if( names_[it->second] != name ) {
58 SC_REPORT_FATAL( sc_core::SC_ID_INTERNAL_ERROR_,
59 "tlm_phase registration failed: duplicate type info" );
60 sc_core::sc_abort();
61 }
62 return it->second;
63 }
64
65 const char* get_name( key_type id ) const
66 {
67 sc_assert( id < names_.size() );
68 return names_[id].c_str();
69 }
70
71private:
72 typedef std::map<sc_type_index, key_type> type_map;
73 typedef std::vector<std::string> name_table;
74
75 type_map ids_;
76 name_table names_;
77
78 tlm_phase_registry()
79 : names_( END_RESP+1 )
80 {
81# define BUILTIN_PHASE(phase) \
82 names_[phase] = #phase
83
84 BUILTIN_PHASE( UNINITIALIZED_PHASE );
85 BUILTIN_PHASE( BEGIN_REQ );
86 BUILTIN_PHASE( END_REQ );
87 BUILTIN_PHASE( BEGIN_RESP );
88 BUILTIN_PHASE( END_RESP );
89 }
90
91}; // class tlm_phase_registry
