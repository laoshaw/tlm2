32class ispex_registry // copied from tlm_gp.cpp
33{
34 typedef unsigned int key_type;
35 typedef std::map<sc_core::sc_type_index, key_type> type_map;
36public:
37 static ispex_registry& instance()
38 {
39 if (!instance_) // don't cleanup registry
40 instance_ = new ispex_registry();
41 return *instance_;
42 }
43
44 unsigned int register_extension(sc_core::sc_type_index type)
45 {
46 type_map::const_iterator it = ids_.find( type );
47
48 if( it == ids_.end() ) { // new extension - generate/store ID
49 type_map::value_type v( type, static_cast<key_type>(ids_.size()) );
50 ids_.insert( v );
51 return v.second;
52 }
53 return it->second;
54 }
55
56 static unsigned int max_num_extensions()
57 { return (instance_) ? instance().ids_.size() : 0; }
58
59private:
60 static ispex_registry* instance_;
61 type_map ids_;
62 ispex_registry() /* = default */ {}
63
64}; // class ispex_registry
65
66ispex_registry* ispex_registry::instance_ = NULL;
67} // anonymous namespace

164void
165instance_specific_extension_container::resize()
166{
167 m_ispex_per_accessor.resize( max_num_ispex_accessors() );
168
169 for (unsigned int i=0; i < m_ispex_per_accessor.size(); ++i) {
170 m_ispex_per_accessor[i] = new instance_specific_extensions_per_accessor(this);
171 m_ispex_per_accessor[i]->resize_extensions();
